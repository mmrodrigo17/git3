git init
inicializar area de trabajo 

git status

git add nombrearchivo.txt
agrego archivo
 git add .
 agrego todos los archivos

git commit
git commit -m "agregado de .gitignore"

git log 
para ver el historial de commit

git checkout -- nombrearchivo.js
para revertir los cambios en un archivo


git diff
para ver las diferencias hechas en los archivos

git branch
muestra los branch

git checkout test
me mueve al branch test


git log
para obtener el <sha1-commit> del commit deseado

git branch <nombre-rama> <sha1-commit>
git branch staging e8f21fa513fe408648d162a5cbce8c4d62733f53
para crear un branch a partir de un commit anterior


git checkout staging
para pararte en el branch deseado
